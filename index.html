Quick Start

import { vlist } from "@floor/vlist"

const list = vlist({
  container: "#list",
  item: {
    height: 48,
    template: (item) => `<div>${item.name}</div>`
  }
}).build()

list.setItems(
  Array.from({ length: 10_000 }, (_, i) => ({
    id: i,
    name: `Item ${i}`
  }))
)
